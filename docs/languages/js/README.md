When a parameter is repeatable, prefix the argument name with an ellipsis e.g. 

```js
function sum(...value) {
  // Implementation goes here
}
sum.pars = ['...value']
```

```mini
sum(3, [])
```

```js
function sum(...values) {
  let result = 0
  for (let value of values) {
    const type_ = type(value)
    switch (type_) {
      case 'number':
        result += value
      case 'array[number]':
        for (let item of value) result += value
      default:
        type_error(type_)
    }
  }
  return result
}
sum.pars = ['...values']
```
